class LogisticRegression:
    def __init__(self,lr=0.01,num_iter=1000,fit_intercept=True,verbose=False):
        self.lr=lr
        self.num_iter=num_iter
        self.fit_intercept=fit_intercept
        self.verbose=verbose
        
    def __add_intercept(self,a):
        intercept=np.ones((a.shape[0],1))
        return np.concatenate((intercept,a),axis=1)
    
    def __sigmoid(self,z):
        return 1/(1+np.exp(-z))
    
    def __loss(self,h,b):
        return (-y*np.log(h)-(1-y)*np.log(1-h)).mean()
    
    def fit(self,a,b):
        if self.fit_intercept:
            a=self.__add_intercept(a)
            
        #weights initialization
        self.theta = np.zeros(a.shape[1])
        
        for i in range(self.num_iter):
            z=np.dot(a,self.theta)
            h=self.__sigmoid(z)
            gradient=np.dot(a.T,(h-b))/b.size
            self.theta-=self.lr*gradient
            
            z=np.dot(a,self.theta)
            h=self.__sigmoid(z)
            loss=self.__loss(h,b)
            
            if(self.verbose==True and i % 10000==0):
                print(f'loss:{loss}\t')
                
    def predict_prob(self,a):
        if self.fit_intercept:
            a= self.__add_intercept(a)
        return self.__sigmoid(np.dot(a,self.theta))
    
    def predict(self,a):
        return self.predict_prob(a).round()

class SVM:
    
    def __init__(self,learning_rate=0.001,lambda_param=100,n_iters=1000):
        self.lr=learning_rate
        self.lambda_param=lambda_param
        self.n_iters=n_iters
        self.w=None
        self.b=None
        
    def fit(self,X,y):
        y_=np.where(y<=0,-1,1)
        n_samples,n_features=X.shape
        
        self.w=np.zeros(n_features)
        self.b=0
        
        for _ in range(self.n_iters):
            for idx,x_i in enumerate(X):
                condition=y_[idx]*(np.dot(x_i,self.w)-self.b)>=1
                if condition:
                    self.w -= self.lr * (2 * self.lambda_param * self.w)
                else:
                    x_i = np.squeeze(np.asarray(x_i))
                    y_ = np.squeeze(np.asarray(y_))
                    self.w -= self.lr * (2 * self.lambda_param * self.w - np.dot(x_i,y_[idx]))
                    self.b -= self.lr - y_[idx]
        return x_i.shape,y_.shape          
    
    def predict(self,X):
        linear_output = np.dot(X,self.w) - self.b
        return np.sign(linear_output)

import numpy as np
def make_meshgrid(x, y, h=.02):
    x_min, x_max = x.min() - 1, x.max() + 1
    y_min, y_max = y.min() - 1, y.max() + 1
    xx, yy = np.meshgrid(np.arange(x_min, x_max, h), np.arange(y_min, y_max, h))
    return xx, yy

def plot_contours(ax, clf, xx, yy, **params):
    Z = clf.predict(np.c_[xx.ravel(), yy.ravel()])
    Z = Z.reshape(xx.shape)
    out = ax.contourf(xx, yy, Z, **params)
    return out
